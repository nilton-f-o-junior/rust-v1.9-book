# Assignment Operators

Used to assign values to variables.

`+=`

```rust
//
let mut num: u8 = 10;

// addition and assignment: +=
num += 1;
println!("10 += 1 = {}", num);
```

`-=`

```rust
//
let mut num: u8 = 10;

// subtraction and assignment: -=
num -= 1;
println!("10 -= 1 = {}", num);
```

`*=`

```rust
//
let mut num: u8 = 10;

// multiplication and assignment: *=
num *= 2;
println!("10 *= 2 = {}", num);
```

`/=`

```rust
//
let mut num: u8 = 20;

// division and assignment: /=
num /= 3;
println!("20 /= 3 = {}", num);
```

`%=`

```rust
//
let mut num: u8 = 6;

// remainder and assignment: %=
num %= 4;
println!("6 %= 4 = {}", num);
```
